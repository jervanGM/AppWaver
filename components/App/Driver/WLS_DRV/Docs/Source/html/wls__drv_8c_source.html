85static void on_got_ip(void *arg,
86 esp_event_base_t event_base,
87 int32_t event_id,
88 void *event_data) {
89 ip_event_got_ip_t *event = (ip_event_got_ip_t *) event_data;
90 last_wifi_event_id = event_id;
91 if (!is_our_netif(TAG, event->esp_netif)) {
92 ESP_LOGW(TAG, "Got IPv4 from another interface \"%s\": ignored",
93 esp_netif_get_desc(event->esp_netif));
94 return;
95 }
96 ESP_LOGI(TAG, "Got IPv4 event: Interface \"%s\" address: " IPSTR,
97 esp_netif_get_desc(event->esp_netif), IP2STR(&event->ip_info.ip));
98 memcpy(&s_ip_addr, &event->ip_info.ip, sizeof(s_ip_addr));
99 ip_obtained = true;
100}

157static esp_err_t connect() {
158 snprintf((char *) wifi_config.sta.ssid,
159 sizeof(wifi_config.sta.ssid), "%s",
160 CONFIG_ANJAY_WIFI_SSID);
161 snprintf((char *) wifi_config.sta.password,
162 sizeof(wifi_config.sta.password), "%s",
163 CONFIG_ANJAY_WIFI_PASSWORD);
164 wifi_config.sta.scan_method = WIFI_ALL_CHANNEL_SCAN;
165
166 ESP_ERROR_CHECK(esp_event_handler_register(WIFI_EVENT,
167 WIFI_EVENT_STA_DISCONNECTED,
168 &on_wifi_disconnect, NULL));
169 ESP_ERROR_CHECK(esp_event_handler_register(IP_EVENT, IP_EVENT_STA_GOT_IP,
170 &on_got_ip, NULL));
171 #ifdef CONFIG_ANJAY_WIFI_CONNECT_IPV6
172 ESP_ERROR_CHECK(
173 esp_event_handler_register(WIFI_EVENT, WIFI_EVENT_STA_CONNECTED,
174 &on_wifi_connect, s_anjay_esp_netif));
175 ESP_ERROR_CHECK(esp_event_handler_register(IP_EVENT, IP_EVENT_GOT_IP6,
176 &on_got_ipv6, NULL));
177 #endif // CONFIG_ANJAY_WIFI_CONNECT_IPV6
178
179 ESP_ERROR_CHECK(esp_wifi_set_storage(WIFI_STORAGE_RAM));
180 ESP_LOGI(TAG, "Connecting to %s...", wifi_config.sta.ssid);
181 ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_STA));
182 ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_STA, &wifi_config));
183 ESP_ERROR_CHECK(esp_wifi_start());
184 esp_wifi_connect();
185
186 //ESP_ERROR_CHECK(esp_register_shutdown_handler(&stop));
187 ESP_LOGI(TAG, "Waiting for IP(s)");
188 // Obtener el tiempo actual en milisegundos
189 uint32_t start_time = esp_log_timestamp();
190
191 // Esperar hasta que se obtenga la dirección IP o se agote el tiempo de espera
192 while (!ip_obtained && (esp_log_timestamp() - start_time < MAX_WAITING_TIME_FOR_IP)) {
193 // Aquí puedes realizar otras tareas si es necesario
194 }
195
196 // Verificar si se ha obtenido la dirección IP o se ha agotado el tiempo de espera
197 if (!ip_obtained) {
198 ESP_LOGE(TAG, "Failed to obtain IP address within the timeout period");
199 return ESP_FAIL;
200 }
201 // iterate over active interfaces, and print out IPs of "our" netifs
202 esp_netif_t *netif = NULL;
203 esp_netif_ip_info_t ip;
204 for (int i = 0; i < esp_netif_get_nr_of_ifs(); ++i) {
205 netif = esp_netif_next(netif);
206 if (is_our_netif(TAG, netif)) {
207 ESP_LOGI(TAG, "Connected to %s", esp_netif_get_desc(netif));
208 ESP_ERROR_CHECK(esp_netif_get_ip_info(netif, &ip));
209
210 ESP_LOGI(TAG, "- IPv4 address: " IPSTR, IP2STR(&ip.ip));
211 #ifdef CONFIG_ANJAY_WIFI_CONNECT_IPV6
212 esp_ip6_addr_t ip6[MAX_IP6_ADDRS_PER_NETIF];
213 int ip6_addrs = esp_netif_get_all_ip6(netif, ip6);
214 for (int j = 0; j < ip6_addrs; ++j) {
215 esp_ip6_addr_type_t ipv6_type =
216 esp_netif_ip6_get_addr_type(&(ip6[j]));
217 ESP_LOGI(TAG, "- IPv6 address: " IPV6STR ", type: %s",
218 IPV62STR(ip6[j]), s_ipv6_addr_types[ipv6_type]);
219 }
220 #endif // CONFIG_ANJAY_WIFI_CONNECT_IPV6
221 }
222 }
223 return ESP_OK;
224}

226static void disconnect(void)
227{
228 // Unregister Wi-Fi event handlers
229 ESP_ERROR_CHECK(esp_event_handler_unregister(
230 WIFI_EVENT, WIFI_EVENT_STA_DISCONNECTED, &on_wifi_disconnect));
231 ESP_ERROR_CHECK(esp_event_handler_unregister(IP_EVENT, IP_EVENT_STA_GOT_IP,
232 &on_got_ip));
233
234 // Unregister IPv6 event handlers if IPv6 is enabled
235 #ifdef CONFIG_ANJAY_WIFI_CONNECT_IPV6
236 ESP_ERROR_CHECK(esp_event_handler_unregister(IP_EVENT, IP_EVENT_GOT_IP6,
237 &on_got_ipv6));
238 ESP_ERROR_CHECK(esp_event_handler_unregister(
239 WIFI_EVENT, WIFI_EVENT_STA_CONNECTED, &on_wifi_connect));
240 #endif // CONFIG_ANJAY_WIFI_CONNECT_IPV6
241
242 // Stop Wi-Fi operations
243 esp_err_t err = esp_wifi_stop();
244 if (err == ESP_ERR_WIFI_NOT_INIT) {
245 return; // Wi-Fi not initialized, nothing to stop
246 }
247 ESP_ERROR_CHECK(err); // Check for errors during Wi-Fi stop
248}

265{
266 ESP_LOGI(TAG, "Wifi initialization");
267 if((nvs_flash_init() != ESP_OK) ||
268 (esp_netif_init() != ESP_OK) ||
269 (esp_event_loop_create_default() != ESP_OK)){
270 return -1;
271 }
272 char *desc;
273
274 wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
275 if(esp_wifi_init(&cfg) != ESP_OK){
276 return -1;
277 }
278
279 esp_netif_inherent_config_t esp_netif_config =
280 ESP_NETIF_INHERENT_DEFAULT_WIFI_STA();
281 // Prefix the interface description with the module TAG
282 // Warning: the interface desc is used in tests to capture actual
283 // connection details (IP, gw, mask)
284 asprintf(&desc, "%s: %s", TAG, esp_netif_config.if_desc);
285 esp_netif_config.if_desc = desc;
286 esp_netif_config.route_prio = 128;
287 s_anjay_esp_netif = esp_netif_create_wifi(WIFI_IF_STA, &esp_netif_config);
288 free(desc);
289 esp_wifi_set_default_wifi_sta_handlers();
290 return 0;
291}
