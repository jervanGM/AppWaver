172void on_ctrl_breakdown()
173{
174 // Fault reason
175 ECtrlTaskStatus_t fault_reason = CTRL_TASK_OK;
176 fault_reason = control_app_check_faults();
177 //Clean the error memory
178 store_error_in_slot(CONTROL_ERROR_SLOT,0);
179 switch (fault_reason)
180 {
181 case CTRL_MINOR_FAULT:
182 // Log minor fault
183 TRACE_WARNING("A minor fault has been produced on controller task");
184 // Set state machine event to previous
185 control_sm_set_st_event(CTRL_STATE_PREV);
186 break;
187
188 case CTRL_MAYOR_FAULT:
189 // Log major fault
190 ASSERT_PANIC(false,"A mayor fault reason has been produced on controller task, reboot application");
191 break;
192 default:
193 // Assert if unknown fault reason
194 ASSERT_PANIC(false,"Unknown fault reason has been produced on controller task");
195 break;
196 }
