59 err += bus_port->write_reg(SHT4x_DEFAULT_ADDR, &cmd, 1);

65 if (serial == 0 || err != BUS_DRV_OK)
66 {
67 // Handle initialization error

158void measure_raw_data(uint8_t *raw_data)
159{
160 uint8_t rxbytes[6] = {0};
161 uint8_t total_data[RAW_DATA_BYTES];
162 uint8_t cmd = SHT4x_NOHEAT_LOWPRECISION;
163 if(raw_data != NULL && bus_port != NULL)
164 {
165 //Read SHTx values with heater disabled.
166 int8_t err = 0;
167 err += bus_port->write_reg(SHT4x_DEFAULT_ADDR,&cmd,1);
168 task_delay(10);
169 err += bus_port->read_reg(SHT4x_DEFAULT_ADDR,rxbytes,6);
170 memcpy(total_data, rxbytes, sizeof(uint8_t) * 6);
171 //Reads X,Y,Z MSB and LSB values
172 err += read_register(LIS3DH_DEFAULT_ADDR, LIS3DH_OUT_X_L, &total_data[7], LIS3DH_READ_DELAY);
173 err += read_register(LIS3DH_DEFAULT_ADDR, LIS3DH_OUT_X_H, &total_data[6], LIS3DH_READ_DELAY);
174 err += read_register(LIS3DH_DEFAULT_ADDR, LIS3DH_OUT_Y_L, &total_data[9], LIS3DH_READ_DELAY);
175 err += read_register(LIS3DH_DEFAULT_ADDR, LIS3DH_OUT_Y_H, &total_data[8], LIS3DH_READ_DELAY);
176 err += read_register(LIS3DH_DEFAULT_ADDR, LIS3DH_OUT_Z_L, &total_data[11], LIS3DH_READ_DELAY);
177 err += read_register(LIS3DH_DEFAULT_ADDR, LIS3DH_OUT_Z_H, &total_data[10], LIS3DH_READ_DELAY);
178 //Reads MSB and LSB from Acceleromenter ADC values
179 err += read_register(LIS3DH_DEFAULT_ADDR, LIS3DH_OUT_ADC1_L, &total_data[13], LIS3DH_READ_DELAY);
180 err += read_register(LIS3DH_DEFAULT_ADDR, LIS3DH_OUT_ADC1_H, &total_data[12], LIS3DH_READ_DELAY);
181 err += read_register(LIS3DH_DEFAULT_ADDR, LIS3DH_OUT_ADC2_L, &total_data[15], LIS3DH_READ_DELAY);
182 err += read_register(LIS3DH_DEFAULT_ADDR, LIS3DH_OUT_ADC2_H, &total_data[14], LIS3DH_READ_DELAY);
183 err += read_register(LIS3DH_DEFAULT_ADDR, LIS3DH_OUT_ADC3_L, &total_data[17], LIS3DH_READ_DELAY);
184 err += read_register(LIS3DH_DEFAULT_ADDR, LIS3DH_OUT_ADC3_H, &total_data[16], LIS3DH_READ_DELAY);
185
186 if(err != BUS_DRV_OK)
187 {
188 store_error_in_slot(BUS_ERROR_SLOT,BUS_DRV_ON_READ_ERROR);
189 memset(raw_data, 0, RAW_DATA_BYTES);
190 }
191 else
192 {
193 memcpy(raw_data, total_data, RAW_DATA_BYTES);
194 }
195 }
196 else
197 {
198 store_error_in_slot(BUS_ERROR_SLOT,BUS_DRV_ON_READ_FATAL_ERROR);
199 TRACE_ERROR("Digital bus raw data buffer is not valid");
200 }

204void write_temp_moist_cmd(ESht4xHeaterCmd_t cmd)
205{
206 uint8_t reg = SHT4x_NOHEAT_LOWPRECISION;
207 int8_t err = BUS_DRV_OK;
208 //Send to SHT sensor heater duration command.
209 if(bus_port != NULL)
210 {
211 switch (cmd)
212 {
213 case SHT4X_NO_HEATER:
214 //Do nothing
215 break;
216 case SHT4X_HIGH_HEATER_1S:
217 reg = SHT4x_HIGHHEAT_1S;
218 err = bus_port->write_reg(LIS3DH_DEFAULT_ADDR,&reg,1);
219 break;
220
221 case SHT4X_HIGH_HEATER_100MS:
222 reg = SHT4x_HIGHHEAT_100MS;
223 err = bus_port->write_reg(LIS3DH_DEFAULT_ADDR,&reg,1);
224 break;
225
226 case SHT4X_MED_HEATER_1S:
227 reg = SHT4x_MEDHEAT_1S;
228 err = bus_port->write_reg(LIS3DH_DEFAULT_ADDR,&reg,1);
229 break;
230
231 case SHT4X_MED_HEATER_100MS:
232 reg = SHT4x_MEDHEAT_100MS;
233 err = bus_port->write_reg(LIS3DH_DEFAULT_ADDR,&reg,1);
234 break;
235
236 case SHT4X_LOW_HEATER_1S:
237 reg = SHT4x_LOWHEAT_1S;
238 err = bus_port->write_reg(LIS3DH_DEFAULT_ADDR,&reg,1);
239 break;
240
241 case SHT4X_LOW_HEATER_100MS:
242 reg = SHT4x_LOWHEAT_100MS;
243 err = bus_port->write_reg(LIS3DH_DEFAULT_ADDR,&reg,1);
244 break;
245
246 default:
247 store_error_in_slot(BUS_ERROR_SLOT,BUS_DRV_HEAT_CMD_INVALID);
248 TRACE_ERROR("Moisture sensor temperature heater command is not valid");
249 break;
250 }
251 }
252 else
253 {
254 err = BUS_DRV_FAIL;
255 }
256
257 if(err != BUS_DRV_OK)
258 {
259 store_error_in_slot(BUS_ERROR_SLOT,BUS_DRV_ON_WRITE_ERROR);
260 }
