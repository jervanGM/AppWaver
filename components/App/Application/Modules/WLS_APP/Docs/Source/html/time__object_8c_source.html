80static time_instance_t *add_instance(time_object_t *obj, anjay_iid_t iid) {
81 assert(find_instance(obj, iid) == NULL);

83 AVS_LIST(time_instance_t) created = AVS_LIST_NEW_ELEMENT(time_instance_t);
