187
188 // Placeholder for error status (currently set to false)
189 bool err = false;
190
191 // Check if previously connected and now an error occurred
192 if (connected_prev && err) {
193 // Transition to offline mode
194 connected_prev = false;
195 anjay_transport_enter_offline(anjay, ANJAY_TRANSPORT_SET_IP);
196 }
197 // Check if previously offline and no error occurred

446static void send_job(avs_sched_t *sched, const void *anjay_ptr) {
447 anjay_t *anjay = *(anjay_t *const *) anjay_ptr;
448 const anjay_ssid_t server_ssid = 1;
449
450 // Allocate new batch builder.
451 anjay_send_batch_builder_t *builder = anjay_send_batch_builder_new();
452
453 if (!builder) {
454 TRACE_ERROR("Failed to allocate batch builder");
455 return;
456 }
457
458 int32_t res = 0;
459
460 plant_object_send(builder,anjay, PLANT_DATA_OBJ);
461 accelerometer_object_send(builder,anjay, ACCELEROMETER_DATA_OBJ);
462
463 // After adding all values, compile our batch for sending.
464 anjay_send_batch_t *batch = anjay_send_batch_builder_compile(&builder);
465
466 if (!batch) {
467 anjay_send_batch_builder_cleanup(&builder);
468 TRACE_ERROR("Batch compile failed");
469 return;
470 }
471
472 // Schedule our send to be run on next `anjay_sched_run()` call.
473 res = (int32_t)anjay_send(anjay, server_ssid, batch, NULL, NULL);
474
475 if (res) {
476 TRACE_ERROR("Failed to send, result:", TO_STRING(res));
477 }
478
479 // After scheduling, we can release our batch.
480 anjay_send_batch_release(&batch);
481
482 // Schedule run of the same function after 10 seconds
483 AVS_SCHED_DELAYED(sched, NULL,
484 avs_time_duration_from_scalar(1, AVS_TIME_S), send_job,
485 &anjay, sizeof(anjay));
486}
