77static int resource_read(anjay_t *anjay,
78 const anjay_dm_object_def_t *const *obj_ptr,
79 anjay_iid_t iid,
80 anjay_rid_t rid,
81 anjay_riid_t riid,
82 anjay_output_ctx_t *ctx) {
83 (void) anjay;
84
85 device_object_t *obj = get_obj(obj_ptr);
86 assert(obj);
87 assert(iid == 0);
88
89 switch (rid) {
90 case RID_MANUFACTURER:
91 assert(riid == ANJAY_ID_INVALID);
92 return anjay_ret_string(ctx, CONFIG_ANJAY_CLIENT_DEVICE_MANUFACTURER);
93
94 case RID_MODEL_NUMBER:
95 assert(riid == ANJAY_ID_INVALID);
96 return anjay_ret_string(ctx, CONFIG_ANJAY_CLIENT_MODEL_NUMBER);
97
98 case RID_SERIAL_NUMBER:
99 assert(riid == ANJAY_ID_INVALID);
100 return anjay_ret_string(ctx, obj->serial_number.value);
101
102 case RID_FIRMWARE_VERSION:
103 assert(riid == ANJAY_ID_INVALID);
104 return anjay_ret_string(ctx, CLIENT_VERSION);
105
106 case RID_ERROR_CODE:
107 assert(riid == 0);
108 return anjay_ret_i32(ctx, 0);
109
110 case RID_DEVICE_CURRENT_TIME:
111 assert(riid == ANJAY_ID_INVALID);
112 return anjay_ret_i64(ctx, obj->current_time);
113
114 case RID_SUPPORTED_BINDING_AND_MODES:
115 assert(riid == ANJAY_ID_INVALID);
116 return anjay_ret_string(ctx, SUPPORTED_BINDING_MODES);
117
118 case RID_SOFTWARE_VERSION:
119 assert(riid == ANJAY_ID_INVALID);
120 return anjay_ret_string(ctx, anjay_get_version());
121
122 default:
123 return ANJAY_ERR_METHOD_NOT_ALLOWED;
124 }
125}

141static int resource_execute(anjay_t *anjay,

199const anjay_dm_object_def_t **device_object_create(void) {
200 device_object_t *obj =
201 (device_object_t *) avs_calloc(1, sizeof(device_object_t));
202 if (!obj) {
203 return NULL;
204 }
205 obj->def = &OBJ_DEF;
206
207 if (get_device_id(&obj->serial_number)) {
208 obj->serial_number.value[0] = '\0';
209 }
210
211 return &obj->def;
212}

227void device_object_update(anjay_t *anjay,
228 const anjay_dm_object_def_t *const *def) {
229 if (!anjay || !def) {
230 return;
231 }
232
233 device_object_t *obj = get_obj(def);
234 const IWlsPort *port = hal_wls_get_port();
235 if (obj->do_reboot) {
236 // This is a bit harsh, but this is the nicest way to ensure
237 // the reboot using the public ESP32 API
238 port->system_abort("Rebooting ...");
239 }
