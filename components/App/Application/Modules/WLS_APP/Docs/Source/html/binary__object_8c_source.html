80static binary_instance_t *add_instance(binary_object_t *obj, anjay_iid_t iid) {
81 assert(find_instance(obj, iid) == NULL);
82
83 AVS_LIST(binary_instance_t) created = AVS_LIST_NEW_ELEMENT(binary_instance_t);
84 if (!created) {
85 return NULL;

87
88 int result = init_instance(created, iid);
89 if (result) {
90 AVS_LIST_CLEAR(&created);
91 return NULL;

94 AVS_LIST(binary_instance_t) *ptr;
95 AVS_LIST_FOREACH_PTR(ptr, &obj->instances) {
96 if ((*ptr)->iid > created->iid) {
97 break;
98 }
99 }
100
101 AVS_LIST_INSERT(ptr, created);
102 return created;

164static int resource_read(anjay_t *anjay,
165 const anjay_dm_object_def_t *const *obj_ptr,

187 return ANJAY_ERR_METHOD_NOT_ALLOWED;
