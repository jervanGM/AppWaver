82static percentage_instance_t *add_instance(percentage_object_t *obj, anjay_iid_t iid) {
83 assert(find_instance(obj, iid) == NULL);
84
85 AVS_LIST(percentage_instance_t) created = AVS_LIST_NEW_ELEMENT(percentage_instance_t);
86 if (!created) {
87 return NULL;
88 }
89
90 int result = init_instance(created, iid);
91 if (result) {
92 AVS_LIST_CLEAR(&created);
93 return NULL;
94 }
95
96 AVS_LIST(percentage_instance_t) *ptr;
97 AVS_LIST_FOREACH_PTR(ptr, &obj->instances) {
98 if ((*ptr)->iid > created->iid) {
99 break;
100 }
101 }
102
103 AVS_LIST_INSERT(ptr, created);
104 return created;
105}

168static int resource_read(anjay_t *anjay,
