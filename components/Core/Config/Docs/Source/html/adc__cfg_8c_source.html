128static bool adc_calibration_ana_init(adc_unit_t unit, adc_channel_t channel, adc_atten_t atten, adc_cali_handle_t *out_handle)
129{
130 adc_cali_handle_t handle = NULL;
131 esp_err_t ret = ESP_FAIL;
132 bool calibrated = false;
133
134#if ADC_CALI_SCHEME_CURVE_FITTING_SUPPORTED
135 if (!calibrated) {
136 ESP_LOGI(TAG, "calibration scheme version is %s", "Curve Fitting");
137 adc_cali_curve_fitting_config_t cali_config = {
138 .unit_id = unit,
139 .chan = channel,
140 .atten = atten,
141 .bitwidth = ADC_BITWIDTH_DEFAULT,
142 };
143 ret = adc_cali_create_scheme_curve_fitting(&cali_config, &handle);
144 if (ret == ESP_OK) {
145 calibrated = true;
146 }
147 }
148#endif
149
150#if ADC_CALI_SCHEME_LINE_FITTING_SUPPORTED
151 if (!calibrated) {
152 ESP_LOGI(TAG, "calibration scheme version is %s", "Line Fitting");
153 adc_cali_line_fitting_config_t cali_config = {
154 .unit_id = unit,
155 .atten = atten,
156 .bitwidth = ADC_BITWIDTH_DEFAULT,
157 };
158 ret = adc_cali_create_scheme_line_fitting(&cali_config, &handle);
159 if (ret == ESP_OK) {
160 calibrated = true;
161 }
162 }
163#endif
164
165 *out_handle = handle;
166 if (ret == ESP_OK) {
167 ESP_LOGI(TAG, "Calibration Success");
168 } else if (ret == ESP_ERR_NOT_SUPPORTED || !calibrated) {
169 ESP_LOGW(TAG, "eFuse not burnt, skip software calibration");
170 } else {
171 ESP_LOGE(TAG, "Invalid arg or no memory");
172 }
173
174 return calibrated;
175}
